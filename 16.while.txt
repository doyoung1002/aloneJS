while (불_표현식) {
  본문
}
다음줄

1. 불_표현식을 확인한다.
2-a. true라면 -> 본문을 실행하고 1번으로 돌아간다.
2-b. false -> 종료

let i = 0;
while (true) {
  alert(`${i}번째 반복입니다`)
  i++; // 무한 반복함
}

for 반복문과 비교

while (i < 10) { // 조건이 중요할 때
  alert(`${i}번째 반복입니다`)
  i++; // 무한 반복함
}

for (let i = 0; i < 10; i++) { // 특정 횟수 만큼, 배열 기반으로 반복을 돌때

}

let i = 0;
while (confirm('계속 진행?')) {
  alert(`${i}번째 반복입니다`)
  i++; 
}

while 반복문을 쓰는 이유
- 결과가 나올 때까지
- 특정 시간이 될 때까지
- 파일을 읽으며 특정 단어를 찾을 때까지
- 등등
